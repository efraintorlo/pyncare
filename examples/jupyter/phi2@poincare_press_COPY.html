def compact_dyn_sys_phi2(init, t=None, modelpars=[]):
        if t is None:
            t = 1.0  # This is used to allows a call from "plot_vector_flow()"
        X = init[0]
        Y = init[1]
        # the model equations
        Z = np.sqrt(1.0 - X**2.0 - Y**2.0)
        A = 3.0 * Y * np.sqrt(X**2.0 + Y**2.0) / Z
        X_dot = Y + X * Y * A
        Y_dot = -X + (Y**2.0 - 1.0) * A
        return [X_dot, Y_dot]
